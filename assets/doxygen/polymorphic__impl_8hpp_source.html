128  static bool exists( std::type_index const & baseIndex, std::type_index const & derivedIndex )
129  {
130  // First phase of lookup - match base type index
131  auto & baseMap = StaticObject<PolymorphicCasters>::getInstance().map;
132  auto baseIter = baseMap.find( baseIndex );
133  if (baseIter == baseMap.end())
134  return false;
135 
136  // Second phase - find a match from base to derived
137  auto & derivedMap = baseIter->second;
138  auto derivedIter = derivedMap.find( derivedIndex );
139  if (derivedIter == derivedMap.end())
140  return false;
141 
142  return true;
143  }

213  struct PolymorphicVirtualCaster : PolymorphicCaster

219  PolymorphicVirtualCaster()
220  {
221  auto & baseMap = StaticObject<PolymorphicCasters>::getInstance().map;
222  auto baseKey = std::type_index(typeid(Base));
223  auto lb = baseMap.lower_bound(baseKey);
224 
225  {
226  auto & derivedMap = baseMap.insert( lb, {baseKey, {}} )->second;
227  auto derivedKey = std::type_index(typeid(Derived));
228  auto lbd = derivedMap.lower_bound(derivedKey);
229  auto & derivedVec = derivedMap.insert( lbd, { std::move(derivedKey), {}} )->second;
230  derivedVec.push_back( this );
231  }
232 
233  // Find all chainable unregistered relations
234  std::map<std::type_index, std::pair<std::type_index, std::vector<PolymorphicCaster const *>>> unregisteredRelations;
235  {
236  auto checkRelation = [](std::type_index const & baseInfo, std::type_index const & derivedInfo)
237  {
238  const bool exists = PolymorphicCasters::exists( baseInfo, derivedInfo );
239  return std::make_pair( exists, exists ? PolymorphicCasters::lookup( baseInfo, derivedInfo, [](){} ) :
240  std::vector<PolymorphicCaster const *>{} );
241  };
242 
243  for( auto baseIt : baseMap )
244  for( auto derivedIt : baseIt.second )
245  {
246  for( auto otherBaseIt : baseMap )
247  {
248  if( baseIt.first == otherBaseIt.first ) // only interested in chained relations
249  continue;
250 
251  // Check if there exists a mapping otherBase -> base -> derived that is shorter than
252  // any existing otherBase -> derived direct mapping
253  auto otherBaseItToDerived = checkRelation( otherBaseIt.first, derivedIt.first );
254  auto baseToDerived = checkRelation( baseIt.first, derivedIt.first );
255  auto otherBaseToBase = checkRelation( otherBaseIt.first, baseIt.first );
256 
257  const size_t newLength = otherBaseToBase.second.size() + baseToDerived.second.size();
258  const bool isShorterOrFirstPath = !otherBaseItToDerived.first || (newLength < derivedIt.second.size());
259 
260  if( isShorterOrFirstPath &&
261  baseToDerived.first &&
262  otherBaseToBase.first )
263  {
264  std::vector<PolymorphicCaster const *> path = otherBaseToBase.second;
265  path.insert( path.end(), baseToDerived.second.begin(), baseToDerived.second.end() );
266 
267  #ifdef CEREAL_OLDER_GCC
268  unregisteredRelations.insert( std::make_pair(otherBaseIt.first,
269  std::pair<std::type_index, std::vector<PolymorphicCaster const *>>{derivedIt.first, std::move(path)}) );
270  #else // NOT CEREAL_OLDER_GCC
271  unregisteredRelations.emplace( otherBaseIt.first,
272  std::pair<std::type_index, std::vector<PolymorphicCaster const *>>{derivedIt.first, std::move(path)} );
273  #endif // NOT CEREAL_OLDER_GCC
274  }
275  } // end otherBaseIt
276  } // end derivedIt
277  } // end chain lookup
278 
279  // Insert chained relations
280  for( auto it : unregisteredRelations )
281  {
282  auto & derivedMap = baseMap.find( it.first )->second;
283  derivedMap[it.second.first] = it.second.second;
284  }
285  }
