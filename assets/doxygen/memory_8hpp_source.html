246  template <class Archive, class T> inline
247  typename std::enable_if<traits::has_load_and_construct<T, Archive>::value, void>::type
248  load( Archive & ar, memory_detail::PtrWrapper<std::shared_ptr<T> &> & wrapper )
249  {
250  auto & ptr = wrapper.ptr;
251 
252  uint32_t id;
253 
254  ar( _CEREAL_NVP("id", id) );
255 
256  if( id & detail::msb_32bit )
257  {
258  // Storage type for the pointer - since we can't default construct this type,
259  // we'll allocate it using std::aligned_storage and use a custom deleter
260  using ST = typename std::aligned_storage<sizeof(T)>::type;
261 
262  // Valid flag - set to true once construction finishes
263  // This prevents us from calling the destructor on
264  // uninitialized data.
265  auto valid = std::make_shared<bool>( false );
266 
267  // Allocate our storage, which we will treat as
268  // uninitialized until initialized with placement new
269  ptr.reset( reinterpret_cast<T *>( new ST() ),
270  [=]( T * t )
271  {
272  if( valid )
273  t->~T();
274 
275  delete reinterpret_cast<ST *>( t );
276  } );
277 
278  // Register the pointer
279  ar.registerSharedPointer( id, ptr );
280 
281  // Perform the actual loading and allocation
282  memory_detail::loadAndConstructSharedPtr( ar, ptr.get(), typename ::cereal::traits::has_shared_from_this<T>::type() );
283 
284  // Mark pointer as valid (initialized)
285  *valid = true;
286  }
287  else
288  ptr = std::static_pointer_cast<T>(ar.getSharedPointer(id));
289  }

336  typename std::enable_if<traits::has_load_and_construct<T, Archive>::value, void>::type
337  load( Archive & ar, memory_detail::PtrWrapper<std::unique_ptr<T, D> &> & wrapper )
338  {
339  uint8_t isValid;
340  ar( _CEREAL_NVP("valid", isValid) );
341 
342  auto & ptr = wrapper.ptr;
343 
344  if( isValid )
345  {
346  // Storage type for the pointer - since we can't default construct this type,
347  // we'll allocate it using std::aligned_storage
348  using ST = typename std::aligned_storage<sizeof(T)>::type;
349 
350  // Allocate storage - note the ST type so that deleter is correct if
351  // an exception is thrown before we are initialized
352  std::unique_ptr<ST> stPtr( new ST() );
353

355  memory_detail::LoadAndConstructLoadWrapper<Archive, T> loadWrapper( reinterpret_cast<T *>( stPtr.get() ) );
356 
357  // Initialize storage

359 
360  // Transfer ownership to correct unique_ptr type
361  ptr.reset( reinterpret_cast<T *>( stPtr.release() ) );
362  }
