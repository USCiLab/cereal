337  void writeName()
338  {
339  NodeType const & nodeType = itsNodeStack.top();
340 
341  // Start up either an object or an array, depending on state
342  if(nodeType == NodeType::StartArray)
343  {
344  itsWriter.StartArray();
345  itsNodeStack.top() = NodeType::InArray;
346  }
347  else if(nodeType == NodeType::StartObject)
348  {
349  itsNodeStack.top() = NodeType::InObject;
350  itsWriter.StartObject();
351  }
352 
353  // Array types do not output names
354  if(nodeType == NodeType::InArray) return;
355 
356  if(itsNextName == nullptr)
357  {
358  std::string name = "value" + std::to_string( itsNameCounter.top()++ ) + "\0";
359  saveValue(name);
360  }
361  else
362  {
363  saveValue(itsNextName);
364  itsNextName = nullptr;
365  }
366  }
367 
369  void makeArray()
370  {
371  itsNodeStack.top() = NodeType::StartArray;
372  }

531  inline void search( const char * searchName )
532  {
533  const auto len = std::strlen( searchName );
534  size_t index = 0;
535  for( auto it = itsMemberItBegin; it != itsMemberItEnd; ++it, ++index )
536  {
537  const auto currentName = it->name.GetString();
538  if( ( std::strncmp( searchName, currentName, len ) == 0 ) &&
539  ( std::strlen( currentName ) == len ) )
540  {
541  itsIndex = index;
542  return;
543  }
544  }
545 
546  throw Exception("JSON Parsing failed - provided NVP (" + std::string(searchName) + ") not found");
547  }
548 
549  private:
550  MemberIterator itsMemberItBegin, itsMemberItEnd;
551  ValueIterator itsValueItBegin;
552  size_t itsIndex;
553  enum Type {Value, Member, Null_} itsType;
554  };

565  inline void search()
566  {

568  if( itsNextName )
